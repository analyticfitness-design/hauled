```php
<?php

namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use App\Http\Requests\CreateEncargoRequest;
use App\Http\Resources\EncargoResource;
use App\Models\Encargo;
use Illuminate\Http\JsonResponse;
use Illuminate\Http\Request;

class EncargoController extends Controller
{
    public function index(Request $request): JsonResponse
    {
        $encargos = Encargo::where('user_id', $request->user()->id)
            ->orderBy('created_at', 'desc')
            ->get();

        return response()->json(['data' => EncargoResource::collection($encargos)]);
    }

    public function show(Request $request, int $id): JsonResponse
    {
        $encargo = Encargo::where('user_id', $request->user()->id)
            ->findOrFail($id);

        return response()->json(['data' => new EncargoResource($encargo)]);
    }

    public function store(CreateEncargoRequest $request): JsonResponse
    {
        $encargo = Encargo::create([
            'user_id' => $request->user()->id,
            'marca' => $request->marca,
            'producto' => $request->producto,
            'talla' => $request->talla,
            'color' => $request->color,
            'link_referencia' => $request->link_referencia,
            'presupuesto' => $request->presupuesto ?? 0,
            'notas_cliente' => $request->notas_cliente,
            'status' => 'pendiente',
        ]);

        return response()->json([
            'data' => new EncargoResource($encargo),
            'message' => '¡Encargo recibido! Te contactaremos pronto con la cotización.',
        ], 201);
    }
}
```

```php
<?php

namespace App\Filament\Resources;

use App\Filament\Resources\EncargoResource\Pages;
use App\Models\Encargo;
use Filament\Forms;
use Filament\Forms\Form;
use Filament\Notifications\Notification;
use Filament\Resources\Resource;
use Filament\Tables;
use Filament\Tables\Table;

class EncargoResource extends Resource
{
    protected static ?string $model = Encargo::class;
    protected static ?string $navigationIcon = 'heroicon-o-inbox-arrow-down';
    protected static ?string $navigationLabel = 'Encargos';
    protected static ?string $modelLabel = 'Encargo';
    protected static ?string $pluralModelLabel = 'Encargos';
    protected static ?int $navigationSort = 5;

    public static function form(Form $form): Form
    {
        return $form->schema([
            Forms\Components\Section::make('Solicitud del Cliente')->schema([
                Forms\Components\TextInput::make('marca')->label('Marca')->disabled(),
                Forms\Components\TextInput::make('producto')->label('Producto')->disabled(),
                Forms\Components\TextInput::make('talla')->label('Talla')->disabled(),
                Forms\Components\TextInput::make('color')->label('Color')->disabled(),
                Forms\Components\TextInput::make('link_referencia')->label('Link referencia')->url()->disabled(),
                Forms\Components\Textarea::make('notas_cliente')->label('Notas del cliente')->disabled()->columnSpanFull(),
            ])->columns(2),

            Forms\Components\Section::make('Gestión Admin')->schema([
                Forms\Components\Select::make('status')
                    ->label('Estado')
                    ->options([
                        'pendiente'  => '⏳ Pendiente',
                        'cotizado'   => '💬 Cotizado',
                        'aprobado'   => '✅ Aprobado',
                        'en_camino'  => '🚚 En camino',
                        'entregado'  => '📦 Entregado',
                        'cancelado'  => '❌ Cancelado',
                    ])
                    ->required()
                    ->live(),

                Forms\Components\TextInput::make('presupuesto')
                    ->label('Presupuesto cotizado (centavos COP)')
                    ->numeric()
                    ->visible(fn($get) => in_array($get('status'), ['cotizado', 'aprobado', 'en_camino', 'entregado'])),

                Forms\Components\TextInput::make('anticipo')
                    ->label('Anticipo requerido (centavos COP)')
                    ->numeric()
                    ->visible(fn($get) => in_array($get('status'), ['cotizado', 'aprobado', 'en_camino', 'entregado'])),

                Forms\Components\Textarea::make('notas_admin')
                    ->label('Notas internas')
                    ->columnSpanFull(),

                Forms\Components\TextInput::make('wa_link')
                    ->label('Link WhatsApp (generado automáticamente)')
                    ->disabled()
                    ->columnSpanFull(),
            ])->columns(2),
        ]);
    }

    public static function table(Table $table): Table
    {
        return $table
            ->columns([
                Tables\Columns\TextColumn::make('id')->label('#')->sortable(),
                Tables\Columns\TextColumn::make('user.name')->label('Cliente')->searchable(),
                Tables\Columns\TextColumn::make('marca')->label('Marca')->searchable(),
                Tables\Columns\TextColumn::make('producto')->label('Producto')->limit(30)->searchable(),
                Tables\Columns\BadgeColumn::make('status')
                    ->label('Estado')
                    ->colors([
                        'warning' => 'pendiente',
                        'info' => 'cotizado',
                        'success' => 'aprobado',
                        'primary' => 'en_camino',
                        'success' => 'entregado',
                        'danger' => 'cancelado',
                    ]),
                Tables\Columns\TextColumn::make('presupuesto')
                    ->label('Presupuesto')
                    ->formatStateUsing(fn($state) => $state ? '$' . number_format($state / 100, 0, ',', '.') : '—'),
                Tables\Columns\TextColumn::make('created_at')->label('Fecha')->dateTime('d/m/Y')->sortable(),
            ])
            ->filters([
                Tables\Filters\SelectFilter::make('status')->options([
                    'pendiente' => 'Pendiente',
                    'cotizado' => 'Cotizado',
                    'aprobado' => 'Aprobado',
                    'en_camino' => 'En camino',
                    'entregado' => 'Entregado',
                    'cancelado' => 'Cancelado',
                ]),
            ])
            ->actions([
                Tables\Actions\EditAction::make()
                    ->after(function (Encargo $record) {
                        // Auto-generar wa_link cuando se cotiza
                        if ($record->status === 'cotizado' && $record->presupuesto > 0) {
                            $whatsapp = config('app.whatsapp_number', '573000000000');
                            $precio = '$' . number_format($record->presupuesto / 100, 0, ',', '.');
                            $anticipo = $record->anticipo
                                ? '$' . number_format($record->anticipo / 100, 0, ',', '.')
                                : null;

                            $mensaje = "Hola {$record->user->name}! 👋\n\n"
                                . "Tu encargo de *{$record->marca} — {$record->producto}* está listo:\n\n"
                                . "💰 Precio: {$precio}\n"
                                . ($anticipo ? "📦 Anticipo: {$anticipo}\n" : '')
                                . "\n¿Confirmamos? 🚀";

                            $waLink = 'https://wa.me/' . $whatsapp . '?text=' . urlencode($mensaje);
                            $record->update(['wa_link' => $waLink]);

                            Notification::make()
                                ->title('wa.me link generado')
                                ->body('El link de WhatsApp fue generado y guardado.')
                                ->success()
                                ->send();
                        }
                    }),

                Tables\Actions\Action::make('open_whatsapp')
                    ->label('Abrir WhatsApp')
                    ->icon('heroicon-o-chat-bubble-left-right')
                    ->color('success')
                    ->url(fn($record) => $record->wa_link)
                    ->openUrlInNewTab()
                    ->visible(fn($record) => (bool) $record->wa_link),
            ])
            ->defaultSort('created_at', 'desc');
    }

    public static function getPages(): array
    {
        return [
            'index' => Pages\ListEncargos::route('/'),
            'edit' => Pages\EditEncargo::route('/{record}/edit'),
        ];
    }
}
```

<template>

            <input v-model="form.talla" type="text" placeholder="Talla (42, L, XL...)" class="form-control" />
          </div>

            <input v-model="form.color" type="text" placeholder="Color" class="form-control" />
          </div>
          <div class="col-md-4">
            <input v-model="form.link_referencia" type="url" placeholder="Link de referencia (opcional)" class="form-control" />
          </div>
          <div class="col-12">
            <textarea v-model="form.notas_cliente" placeholder="Notas adicionales..." rows="3" class="form-control"></textarea>
          </div>
          <div class="col-12">
            <p v-if="error" class="text-danger">{{ error }}</p>
            <button type="submit" :disabled="loading" class="hauled-btn-primary">
              {{ loading ? 'Enviando...' : 'Solicitar Encargo' }}
            </button>
          </div>
        </div>
      </form>

      <div v-else class="encargo-success">
        <h3>¡Encargo recibido! 🎉</h3>
        <p>Te contactaremos con la cotización por WhatsApp.</p>
        <nuxt-link to="/profile" class="hauled-btn-outline">Ver mis encargos</nuxt-link>
      </div>
    </div>
  </section>
</template>

<script setup lang="ts">
import { toast } from 'vue3-toastify'

const authStore = useAuthStore()
const { apiFetch } = useApi()

const submitted = ref(false)
const loading = ref(false)
const error = ref('')

const form = ref({
  marca: '',
  producto: '',
  talla: '',
  color: '',
  link_referencia: '',
  notas_cliente: '',
})

const handleSubmit = async () => {
  if (!authStore.isAuthenticated) {
    await navigateTo('/login')
    return
  }

  loading.value = true
  error.value = ''

  try {
    await apiFetch('/encargos', {
      method: 'POST',
      body: form.value,
    })
    submitted.value = true
    toast.success('¡Encargo enviado! Te contactamos pronto.')
  } catch (err: any) {
    error.value = err?.data?.message ?? 'Error al enviar el encargo.'
    toast.error(error.value)
  } finally {
    loading.value = false
  }
}
</script>
