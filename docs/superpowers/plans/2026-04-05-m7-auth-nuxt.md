```typescript
// app/pinia/useAuthStore.ts
export const useAuthStore = defineStore('auth', () => {
  const token = ref<string | null>(null)
  const user = ref<any | null>(null)

  const isAuthenticated = computed(() => !!token.value)
  const isAdmin = computed(() => user.value?.role === 'admin')

  const { apiFetch } = useApi()

  function loadFromStorage() {
    if (import.meta.client) {
      token.value = localStorage.getItem('hauled_token')
      const savedUser = localStorage.getItem('hauled_user')
      if (savedUser) user.value = JSON.parse(savedUser)
    }
  }

  function saveToStorage() {
    if (import.meta.client) {
      if (token.value) {
        localStorage.setItem('hauled_token', token.value)
        localStorage.setItem('hauled_user', JSON.stringify(user.value))
      } else {
        localStorage.removeItem('hauled_token')
        localStorage.removeItem('hauled_user')
      }
    }
  }

  async function login(email: string, password: string) {
    const res = await apiFetch<{ user: any; token: string }>('/auth/login', {
      method: 'POST',
      body: { email, password },
    })
    token.value = res.token
    user.value = res.user
    saveToStorage()
    return res.user
  }

  async function register(name: string, email: string, password: string, password_confirmation: string, phone?: string) {
    const res = await apiFetch<{ user: any; token: string }>('/auth/register', {
      method: 'POST',
      body: { name, email, password, password_confirmation, phone },
    })
    token.value = res.token
    user.value = res.user
    saveToStorage()
    return res.user
  }

  async function logout() {
    if (token.value) {
      try {
        await apiFetch('/auth/logout', {
          method: 'POST',
          headers: { Authorization: `Bearer ${token.value}` },
        })
      } catch {
        // ignorar error de red en logout
      }
    }
    token.value = null
    user.value = null
    saveToStorage()
  }

  async function fetchMe() {
    if (!token.value) return
    try {
      const res = await apiFetch<{ user: any }>('/auth/me', {
        headers: { Authorization: `Bearer ${token.value}` },
      })
      user.value = res.user
      saveToStorage()
    } catch {
      // token inválido — limpiar sesión
      token.value = null
      user.value = null
      saveToStorage()
    }
  }

  return {
    token,
    user,
    isAuthenticated,
    isAdmin,
    loadFromStorage,
    login,
    register,
    logout,
    fetchMe,
  }
})
```

Reemplazar el contenido de `app/composables/useApi.ts`:
```typescript
export const useApi = () => {
  const config = useRuntimeConfig()
  const baseURL = config.public.apiBaseUrl

  const apiFetch = <T>(url: string, options: Record<string, any> = {}) => {
    // Intentar leer token del store (solo en cliente)
    let authHeader: Record<string, string> = {}
    if (import.meta.client) {
      const token = localStorage.getItem('hauled_token')
      if (token) {
        authHeader = { Authorization: `Bearer ${token}` }
      }
    }

    return $fetch<T>(`${baseURL}${url}`, {
      headers: {
        'Accept': 'application/json',
        'Content-Type': 'application/json',
        ...authHeader,
        ...options.headers,
      },
      ...options,
    })
  }

  return { apiFetch, baseURL }
}
```

<script setup lang="ts">
import { ref } from 'vue'
import { toast } from 'vue3-toastify'

const authStore = useAuthStore()
const router = useRouter()

const name = ref('')
const email = ref('')
const phone = ref('')
const password = ref('')
const password_confirmation = ref('')
const loading = ref(false)
const error = ref('')

const handleRegister = async () => {
  if (password.value !== password_confirmation.value) {
    error.value = 'Las contraseñas no coinciden.'
    return
  }

  loading.value = true
  error.value = ''

  try {
    await authStore.register(name.value, email.value, password.value, password_confirmation.value, phone.value)
    toast.success('¡Cuenta creada! Bienvenido a HAULED.')
    router.push('/profile')
  } catch (err: any) {
    const errors = err?.data?.errors
    if (errors) {
      error.value = Object.values(errors).flat().join('. ')
    } else {
      error.value = err?.data?.message ?? 'Error al crear la cuenta.'
    }
    toast.error(error.value)
  } finally {
    loading.value = false
  }
}
</script>
